// –ö–æ–Ω—Å—Ç–∞–Ω—Ç—ã
const int ROI_Y_START = 11;
const int ROI_Y_END = 102;
const int ROI_X_START = 16;
const int ROI_X_END = 144;
const int MIN_LINE_WIDTH = 5;
const int MAX_LINE_WIDTH = 60;
const int THRESHOLD = 30;

const int scan_lines[] = {11, 19, 27, 35, 43, 51, 60, 68, 76, 84, 92, 101};
const int NUM_SCAN_LINES = 12;

struct LineSegment {
    int left;
    int right;
    float center;
};

// –î–µ—Ç–µ–∫—Ü–∏—è —Å–µ–≥–º–µ–Ω—Ç–æ–≤ –ª–∏–Ω–∏–∏ –Ω–∞ –æ–¥–Ω–æ–π —Å–∫–∞–Ω–ª–∞–π–Ω–µ
int detect_line_segments(uint8_t* white_bg_row, uint8_t* current_row, 
                        int row_length, LineSegment* segments, int max_segments) {
    int num_segments = 0;
    bool in_segment = false;
    int segment_start = 0;
    
    for (int i = 0; i < row_length; i++) {
        int16_t diff = white_bg_row[i] - current_row[i];
        bool is_line = (diff > THRESHOLD);
        
        if (is_line && !in_segment) {
            // –ù–∞—á–∞–ª–æ —Å–µ–≥–º–µ–Ω—Ç–∞
            in_segment = true;
            segment_start = i;
        } else if (!is_line && in_segment) {
            // –ö–æ–Ω–µ—Ü —Å–µ–≥–º–µ–Ω—Ç–∞
            in_segment = false;
            int width = i - segment_start;
            
            if (width >= MIN_LINE_WIDTH && width <= MAX_LINE_WIDTH) {
                if (num_segments < max_segments) {
                    segments[num_segments].left = segment_start;
                    segments[num_segments].right = i - 1;
                    segments[num_segments].center = (segment_start + i - 1) / 2.0;
                    num_segments++;
                }
            }
        }
    }
    
    // –ü—Ä–æ–≤–µ—Ä–∏—Ç—å –ø–æ—Å–ª–µ–¥–Ω–∏–π —Å–µ–≥–º–µ–Ω—Ç
    if (in_segment) {
        int width = row_length - segment_start;
        if (width >= MIN_LINE_WIDTH && width <= MAX_LINE_WIDTH) {
            if (num_segments < max_segments) {
                segments[num_segments].left = segment_start;
                segments[num_segments].right = row_length - 1;
                segments[num_segments].center = (segment_start + row_length - 1) / 2.0;
                num_segments++;
            }
        }
    }
    
    return num_segments;
}

// –ö–ª–∞—Å—Å–∏—Ñ–∏–∫–∞—Ü–∏—è
const char* classify_scenario(float* centers, int num_centers, int roi_width) {
    if (num_centers == 0) {
        return "white_field";
    }
    
    // –î–µ—Ç–µ–∫—Ü–∏—è terminate (–º–∞–ª–æ —Å–∫–∞–Ω–ª–∞–π–Ω–æ–≤)
    float detection_ratio = (float)num_centers / NUM_SCAN_LINES;
    if (detection_ratio < 0.4) {
        return "terminate";
    }
    
    // –î–µ—Ç–µ–∫—Ü–∏—è –ø–æ–≤–æ—Ä–æ—Ç–∞
    if (num_centers >= 4) {
        int n_third = num_centers / 3;
        if (n_third < 1) n_third = 1;
        
        // –í—ã—á–∏—Å–ª–∏—Ç—å —Å—Ä–µ–¥–Ω–∏–µ –¥–ª—è –Ω–∏–∂–Ω–∏—Ö –∏ –≤–µ—Ä—Ö–Ω–∏—Ö
        float bottom_sum = 0.0, top_sum = 0.0;
        for (int i = 0; i < n_third; i++) {
            bottom_sum += centers[i];
        }
        for (int i = num_centers - n_third; i < num_centers; i++) {
            top_sum += centers[i];
        }
        
        float bottom_avg = bottom_sum / n_third;
        float top_avg = top_sum / n_third;
        float position_drift = (top_avg - bottom_avg) / roi_width;
        
        if (position_drift > 0.08) {
            return "left";
        } else if (position_drift < -0.08) {
            return "right";
        }
    }
    
    return "straight";
}

// –û—Å–Ω–æ–≤–Ω–æ–π —Ü–∏–∫–ª
void loop() {
    camera_fb_t* fb = esp_camera_fb_get();
    
    float centers[NUM_SCAN_LINES];
    int num_centers = 0;
    int roi_width = ROI_X_END - ROI_X_START;
    
    // –î–ª—è –∫–∞–∂–¥–æ–π —Å–∫–∞–Ω–∏—Ä—É—é—â–µ–π –ª–∏–Ω–∏–∏
    for (int i = 0; i < NUM_SCAN_LINES; i++) {
        int y = scan_lines[i];
        
        // –ò–∑–≤–ª–µ—á—å —Å—Ç—Ä–æ–∫–∏
        uint8_t white_bg_row[roi_width];
        uint8_t current_row[roi_width];
        
        for (int x = 0; x < roi_width; x++) {
            int idx = y * 160 + (ROI_X_START + x);
            white_bg_row[x] = white_bg[idx];
            current_row[x] = fb->buf[idx];
        }
        
        // –î–µ—Ç–µ–∫—Ü–∏—è —Å–µ–≥–º–µ–Ω—Ç–æ–≤
        LineSegment segments[10];
        int num_segments = detect_line_segments(white_bg_row, current_row, 
                                               roi_width, segments, 10);
        
        // –ë–µ—Ä–µ–º —Å–∞–º—ã–π –±–æ–ª—å—à–æ–π —Å–µ–≥–º–µ–Ω—Ç
        if (num_segments > 0) {
            LineSegment* largest = &segments[0];
            for (int j = 1; j < num_segments; j++) {
                int current_width = segments[j].right - segments[j].left;
                int largest_width = largest->right - largest->left;
                if (current_width > largest_width) {
                    largest = &segments[j];
                }
            }
            centers[num_centers++] = largest->center;
        }
    }
    
    // –ö–ª–∞—Å—Å–∏—Ñ–∏–∫–∞—Ü–∏—è
    const char* scenario = classify_scenario(centers, num_centers, roi_width);
    
    // –£–ø—Ä–∞–≤–ª–µ–Ω–∏–µ
    if (strcmp(scenario, "terminate") == 0) {
        stop_robot();
    } else if (strcmp(scenario, "left") == 0) {
        turn_left();
    } else if (strcmp(scenario, "right") == 0) {
        turn_right();
    } else {
        go_straight();
    }
    
    esp_camera_fb_return(fb);
}
